jCarousel

Gotten from:

http://sorgalla.com/jcarousel/

Concept Behind jCarousel


  • A jQuery plugin
  • Used for organizing lists of elements
  • Commonly used for images
  • Similar to Foundation's "Orbit"

Required Files

  • The "jcarousel.js" file
  • Latest jQuery
  • HTML and CSS of course
  • main.js

main.js

						

(function($){
	$(function() {

	//Initialize the JCarousel
   	$('.jcarousel').jcarousel({

	//Wrap around to first or last list item
  	wrap:"both"
   	});

   		 //Previous Button
        $('.jcarousel-control-prev')
        	//When clicked, move back 1 list item
            .jcarouselControl({
                target: '-=1'
            });

          //Next Button
        $('.jcarousel-control-next')
        	//When clicked, move forward one list item
            .jcarouselControl({
                target: '+=1'
            });

          //Pagination
        $('.jcarousel-pagination')
        	//Which dot gets highlighted
            .on('jcarouselpagination:active', 'a', function() {
                $(this).addClass('active');
            })
            .on('jcarouselpagination:inactive', 'a', function() {
                $(this).removeClass('active');
            })

            .jcarouselPagination();
    });
 })(jQuery);
